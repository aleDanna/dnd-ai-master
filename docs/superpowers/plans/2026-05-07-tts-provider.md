/**
 * Browser-safe TTS voice constants and provider helpers. Both server and
 * client code can import this file (zero runtime deps on `pg` / Node-only modules).
 *
 * Replaces the older `src/lib/tts-voices.ts` (single-provider). Use the
 * provider-aware helpers below for new code; `isValidAnyVoice` is kept only
 * for the rare case where the provider context isn't available yet.
 */

export type TtsProvider = 'openai' | 'gemini';

export const OPENAI_VOICES = [
  'alloy', 'ash', 'ballad', 'coral', 'echo',
  'fable', 'onyx', 'nova', 'sage', 'shimmer',
] as const;
export type OpenAIVoice = (typeof OPENAI_VOICES)[number];

export const GEMINI_VOICES = [
  'Kore', 'Aoede', 'Puck', 'Charon', 'Fenrir',
  'Leda', 'Orus', 'Zephyr', 'Iapetus', 'Sadachbia',
] as const;
export type GeminiVoice = (typeof GEMINI_VOICES)[number];

export type TtsVoice = OpenAIVoice | GeminiVoice;

export const TTS_PROVIDERS: readonly TtsProvider[] = ['openai', 'gemini'] as const;

const DEFAULT_VOICE: Record<TtsProvider, TtsVoice> = {
  openai: 'onyx',
  gemini: 'Kore',
};

export function voicesFor(provider: TtsProvider): readonly TtsVoice[] {
  return provider === 'gemini' ? GEMINI_VOICES : OPENAI_VOICES;
}

export function defaultVoice(provider: TtsProvider): TtsVoice {
  return DEFAULT_VOICE[provider];
}

export function isValidTtsProvider(v: unknown): v is TtsProvider {
  return typeof v === 'string' && (TTS_PROVIDERS as readonly string[]).includes(v);
}

export function isValidVoice(provider: TtsProvider, voice: unknown): voice is TtsVoice {
  if (typeof voice !== 'string') return false;
  return (voicesFor(provider) as readonly string[]).includes(voice);
}

/** Validates against any provider's voices. Used by code paths that don't
 *  know the provider yet (e.g. fast-path PUT validation). */
export function isValidAnyVoice(voice: unknown): voice is TtsVoice {
  return typeof voice === 'string'
    && ((OPENAI_VOICES as readonly string[]).includes(voice)
      || (GEMINI_VOICES as readonly string[]).includes(voice));
}
```

/** Build a 24kHz mono PCM L16 buffer of `durationSec` of a sine wave at `freq` Hz. */
function makeSinePcm(durationSec: number, freq = 440, sampleRate = 24000): Buffer {
  const totalSamples = Math.floor(durationSec * sampleRate);
  const buf = Buffer.alloc(totalSamples * 2);
  for (let i = 0; i < totalSamples; i++) {
    const sample = Math.round(0.3 * 32767 * Math.sin((2 * Math.PI * freq * i) / sampleRate));
    buf.writeInt16LE(sample, i * 2);
  }
  return buf;
}

import lamejs from 'lamejs';

/**
 * Encode a raw PCM L16 little-endian buffer to MP3.
 *
 * Why: Gemini TTS returns PCM L16 24kHz mono; OpenAI returns MP3 directly.
 * Cache uniformity (`audio/mpeg` everywhere) requires server-side conversion.
 *
 * Endianness assumption: Vercel Linux x64/ARM is little-endian, matching
 * Gemini's PCM emission. If the runtime ever ships big-endian, this needs
 * a byte-swap pass — guard with `os.endianness()` if that day comes.
 */

export interface PcmInput {
  sampleRate: 16000 | 22050 | 24000 | 44100 | 48000;
  channels: 1 | 2;
  /** Optional kbps override; default 128. */
  bitRate?: number;
}

export function encodePcmToMp3(pcm: Buffer, opts: PcmInput): ArrayBuffer {
  const { sampleRate, channels, bitRate = 128 } = opts;
  // Reinterpret the byte buffer as Int16Array (LE).
  const samples = new Int16Array(
    pcm.buffer,
    pcm.byteOffset,
    Math.floor(pcm.byteLength / 2),
  );
  const encoder = new lamejs.Mp3Encoder(channels, sampleRate, bitRate);
  const blockSize = 1152; // lamejs requirement
  const chunks: Uint8Array[] = [];
  for (let i = 0; i < samples.length; i += blockSize) {
    const slice = samples.subarray(i, i + blockSize);
    const buf = encoder.encodeBuffer(slice);
    if (buf.length > 0) chunks.push(new Uint8Array(buf));
  }
  const flush = encoder.flush();
  if (flush.length > 0) chunks.push(new Uint8Array(flush));
  const total = chunks.reduce((n, c) => n + c.byteLength, 0);
  const out = new Uint8Array(total);
  let off = 0;
  for (const c of chunks) {
    out.set(c, off);
    off += c.byteLength;
  }
  // Slice produces a tight ArrayBuffer view (drops any trailing capacity).
  return out.buffer.slice(out.byteOffset, out.byteOffset + out.byteLength);
}
```

import OpenAI from 'openai';
import type { OpenAIVoice } from './voices';

/**
 * OpenAI TTS backend. Extracted from the legacy single-file `src/ai/tts.ts`.
 *
 * Override via env:
 *   OPENAI_TTS_MODEL — default 'gpt-4o-mini-tts'
 */

const TTS_MODEL = process.env.OPENAI_TTS_MODEL ?? 'gpt-4o-mini-tts';

let _client: OpenAI | null = null;
let _override: OpenAI | null = null;

function getClient(): OpenAI {
  if (_override) return _override;
  if (_client) return _client;
  const apiKey = process.env.OPENAI_API_KEY;
  if (!apiKey) throw new Error('OPENAI_API_KEY is not set');
  _client = new OpenAI({ apiKey });
  return _client;
}

/** Test seam: inject a mock client; pass null to clear. */
export function __setOpenAIClientForTest(mock: OpenAI | null): void {
  _override = mock;
}

export interface SynthesizeOpenAIInput {
  text: string;
  voice: OpenAIVoice;
  model?: string;
}

export async function synthesizeSpeechOpenAI(input: SynthesizeOpenAIInput): Promise<ArrayBuffer> {
  if (!input.text.trim()) throw new Error('tts: empty input');
  const client = getClient();
  const response = await client.audio.speech.create({
    model: input.model ?? TTS_MODEL,
    voice: input.voice,
    input: input.text,
    response_format: 'mp3',
  });
  return await response.arrayBuffer();
}
```

import { GoogleGenAI } from '@google/genai';
import type { GeminiVoice } from './voices';
import { encodePcmToMp3 } from './pcm-to-mp3';

/**
 * Gemini TTS backend. Returns MP3 bytes by re-encoding Gemini's PCM L16
 * 24kHz mono output via lamejs — the cache and the API response shape
 * stay identical across providers.
 *
 * Override via env:
 *   GEMINI_TTS_MODEL — default 'gemini-2.5-flash-preview-tts'
 */

const TTS_MODEL = process.env.GEMINI_TTS_MODEL ?? 'gemini-2.5-flash-preview-tts';

let _client: GoogleGenAI | null = null;
let _override: GoogleGenAI | null = null;

function getClient(): GoogleGenAI {
  if (_override) return _override;
  if (_client) return _client;
  const apiKey = process.env.GEMINI_API_KEY;
  if (!apiKey) throw new Error('GEMINI_API_KEY is not set');
  _client = new GoogleGenAI({ apiKey });
  return _client;
}

/** Test seam: inject a mock client; pass null to clear. */
export function __setGeminiClientForTest(mock: GoogleGenAI | null): void {
  _override = mock;
}

export interface SynthesizeGeminiInput {
  text: string;
  voice: GeminiVoice;
  model?: string;
}

export async function synthesizeSpeechGemini(input: SynthesizeGeminiInput): Promise<ArrayBuffer> {
  if (!input.text.trim()) throw new Error('tts: empty input');
  const client = getClient();
  const response = await client.models.generateContent({
    model: input.model ?? TTS_MODEL,
    contents: [{ role: 'user', parts: [{ text: input.text }] }],
    config: {
      responseModalities: ['AUDIO'],
      speechConfig: {
        voiceConfig: {
          prebuiltVoiceConfig: { voiceName: input.voice },
        },
      },
    },
  });
  const parts = response.candidates?.[0]?.content?.parts ?? [];
  const inline = parts.find((p) => p.inlineData?.data)?.inlineData;
  if (!inline?.data) throw new Error('tts: gemini returned no audio');
  // PCM L16 little-endian, 24kHz, mono — encode to MP3 to match cache shape.
  const pcm = Buffer.from(inline.data, 'base64');
  return encodePcmToMp3(pcm, { sampleRate: 24000, channels: 1 });
}
```

import type { TtsProvider, TtsVoice, OpenAIVoice, GeminiVoice } from './voices';
import { isValidVoice } from './voices';
import { synthesizeSpeechOpenAI } from './openai';
import { synthesizeSpeechGemini } from './gemini';

export type { TtsProvider, TtsVoice, OpenAIVoice, GeminiVoice } from './voices';

export interface SynthesizeInput {
  text: string;
  provider: TtsProvider;
  voice: TtsVoice;
}

/** Dispatch synthesis to the right backend. Validates voice⇄provider match. */
export async function synthesizeSpeech(input: SynthesizeInput): Promise<ArrayBuffer> {
  if (!isValidVoice(input.provider, input.voice)) {
    throw new Error(`tts: voice "${input.voice}" not valid for provider "${input.provider}"`);
  }
  if (input.provider === 'gemini') {
    return synthesizeSpeechGemini({ text: input.text, voice: input.voice as GeminiVoice });
  }
  return synthesizeSpeechOpenAI({ text: input.text, voice: input.voice as OpenAIVoice });
}
